Start Program

    Set x to 10 int

    Define Function calculate (a int, b int) {
        Set x to 3 int
        Set powerResult to pow(a, b)
        Set sinValue to sin(a)
        Set outerX to parent::x

        Display "Inner x:", x
        Display "Outer x:", outerX
        Display "Power result:", powerResult
        Display "Sin of a:", sinValue
        Display "Sum:", a + b
    } End Function

    calculate(2, 3)
    
End Program
