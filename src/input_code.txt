Start Program
    Set x to 4 int
    Set y to 5 int
    Display (y + x * 5) * 2
    Display x + y, "Sum: ", (x + y) * 2


    Set greeting to "Hello" string
    Set name to "Alice" string
    Display greeting + " " + name

    Set a to 5 int
    Set b to 10 int
    Set c to 6 int
    Set isLess to a < b bool
    Set isMore to a > b bool
    Set result to isLess and not isMore bool
    Display result

    If (a < b and a < c)
        Display "a is less than b and c"
    Else If (a == b)
        Display "a is equal to b"
    Else
        Display "a is greater than b"


    Set i to 0 int
    While (i <= 3)
    {
        Display i
        Set i to i + 1 int
    }

    Set i to 0 int
    For (Set j to 10 int; j > 0; Set j to j - 1 int)
    {
        If (j / 2 == 0)
            Display "Even:", j
        Else
            Display "Odd:", j

        Set i to i + 1 int
    }
    Display "Loop ran", i, "times"


    Define Function addNumbers (num1 int, num2 int){
        Set result to num1 + num2 int
        Return result
    }End Function

    Call addNumbers (5, 3)

    Define Function printMessage (message string){ 
        Return message
    }End Function

    Call printMessage ("Hello, World!")


    
End Program
